class Solution {
public:
  int maxArea(vector<int> &height) {
    int size = height.size();

    int maxCapaticy = 0;
    for (int i = 0; i < size; ++i) {
      for (int j = i + 1; j < size; ++j) {
        int length = j - i;
        int high = min(height[i], height[j]);
        int tmp = length * high;
        if (tmp > maxCapaticy) {
          maxCapaticy = tmp;
        }
      }
    }
    return maxCapaticy;
  }
};

class Solution {
public:
  int maxArea(vector<int> &height) {

    int maxCapacity = 0;
    while (low < high) {

      int tmp;
      if (height[low] > height[high]) {

    }
    return maxCapacity;
  }
};
